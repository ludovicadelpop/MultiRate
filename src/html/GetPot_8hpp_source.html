00707     {

00709

00712             continue;
00713         }

00716         if ( arg.length() > 1 && arg[0] == '[' && arg[arg.length() - 1] == ']' )

00797 GetPot::__read_in_stream (std::istream& istr)
00798 {
00799     STRING_VECTOR  brute_tokens;

00801     {
00802         __skip_whitespace (istr);
00803         const std::string Token = __get_next_token (istr);
00804         if ( Token.length() == 0 || Token[0] == EOF)
00805         {
00806             break;
00807         }
00808         brute_tokens.push_back (Token);
00809     }
00810 
00811     // -- reduce expressions of token1'='token2 to a single
00812     //    string 'token1=token2'

00814     // -- arguments preceded by something like '[' name ']' (section)
00815     //    produce a second copy of each argument with a prefix '[name]argument'

00817     unsigned i2 = 1;

00822     {
00823         const std::string& SRef = brute_tokens[i1];
00824         // 1) concatinate 'abcdef' '=' 'efgasdef' to 'abcdef=efgasdef'
00825         // note: java.lang.String: substring(a,b) = from a to b-1
00826         //        C++ string:      substr(a,b)    = from a to a + b
00827         if ( i2 < brute_tokens.size() && brute_tokens[i2] == "=" )
00828         {
00829             if ( i3 >= brute_tokens.size() )
00830             {
00831                 arglist.push_back (brute_tokens[i1] + brute_tokens[i2]);

00833             else

00854 GetPot::__skip_whitespace (std::istream& istr)
00855 // find next non-whitespace while deleting comments
00856 {
00857     int tmp = istr.get();
00858     do
00859     {
00860         // -- search a non whitespace
00861         while ( isspace (tmp) )
00862         {
00863             tmp = istr.get();
00864             if ( ! istr )
00865             {
00866                 return;
00867             }
00868         }
00869 
00870         // -- look if characters match the comment starter string
00871         unsigned    i = 0;
00872         for (; i < _comment_start.length() ; ++i)
00873         {
00874             if ( tmp != _comment_start[i] )
00875             {
00876                 // NOTE: Due to a 'strange behavior' in Microsoft's streaming lib we do
00877                 // a series of unget()s instead a quick seek. See
00878                 // http://sourceforge.net/tracker/index.php?func=detail&aid=1545239&group_id=31994&atid=403915
00879                 // for a detailed discussion.

00881                 // -- one step more backwards, since 'tmp' already at non-whitespace
00882                 do
00883                 {
00884                     istr.unget();
00885                 }
00886                 while ( i-- != 0 );
00887                 return;
00888             }
00889             tmp = istr.get();
00890             if ( ! istr )
00891             {
00892                 istr.unget();
00893                 return;
00894             }
00895         }
00896         // 'tmp' contains last character of _comment_starter
00897 
00898         // -- comment starter found -> search for comment ender
00899         unsigned match_no = 0;
00900         while (1 + 1 == 2)
00901         {
00902             tmp = istr.get();
00903             if ( ! istr )
00904             {
00905                 istr.unget();
00906                 return;
00907             }

00910             {
00911                 match_no++;
00912                 if ( match_no == _comment_end.length() )
00913                 {
00914                     istr.unget();
00915                     break; // shuffle more whitespace, end of comment found
00916                 }
00917             }
00918             else
00919             {
00920                 match_no = 0;
00921             }

00933 // get next concatinates string token. consider quotes that embrace
00934 // whitespaces

00938     int    last_letter = 0;

00940     {
00941         last_letter = tmp;

00947         }

00949         {
00950             // QUOTES: un-backslashed quotes => it's a string
00951             token += __get_string (istr);

00964         }

00966         {

01044                                  STRING_VECTOR& section_stack)
01045 {
01046     std::string sname = Section;
01047     //  1) subsection of actual section ('./' prefix)
01048     if ( sname.length() >= 2 && sname.substr (0, 2) == "./" )
01049     {
01050         sname = sname.substr (2);
01051     }
01052     //  2) subsection of parent section ('../' prefix)
01053     else if ( sname.length() >= 3 && sname.substr (0, 3) == "../" )
01054     {
01055         do
01056         {
01057             if ( section_stack.end() != section_stack.begin() )
01058             {
01059                 section_stack.pop_back();
01060             }
01061             sname = sname.substr (3);
01062         }
01063         while ( sname.substr (0, 3) == "../" );
01064     }
01065     // 3) subsection of the root-section
01066     else
01067     {
01068         section_stack.erase (section_stack.begin(), section_stack.end() );
01069         // [] => back to root section
01070     }
01071 
01072     if ( sname != "" )
01073     {
01074         // parse section name for 'slashes'
01075         unsigned i = 0;
01076         while ( i < sname.length() )
01077         {
01078             if ( sname[i] == '/' )
01079             {
01080                 section_stack.push_back (sname.substr (0, i) );

01082                 {

02359 // (*) dollar bracket expressions (DBEs) ------------------------------------
02360 //
02361 //     1) Entry Function: __DBE_expand_string()
02362 //        Takes a string such as
02363 //
02364 //          "${+ ${x} ${y}}   Subject-${& ${section} ${subsection}}:   ${title}"
02365 //
02366 //        calls __DBE_expand() for each of the expressions
02367 //
02368 //           ${+ ${x} ${y}}
02369 //           ${& ${section} ${subsection}}
02370 //           ${Title}
02371 //
02372 //        and returns the string
02373 //
02374 //          "4711 Subject-1.01:   Mit den Clowns kamen die Schwaene"
02375 //
02376 //        assuming that
02377 //            x          = "4699"
02378 //            y          = "12"
02379 //            section    = "1."
02380 //            subsection = "01"
02381 //            title      = "Mit den Clowns kamen die Schwaene"
02382 //
02383 //      2) __DBE_expand():
02384 //
02385 //           checks for the command, i.e. the 'sign' that follows '${'
02386 //           divides the argument list into sub-expressions using
02387 //           __DBE_get_expr_list()
02388 //
02389 //           ${+ ${x} ${y}}                 -> "${x}"  "${y}"
02390 //           ${& ${section} ${subsection}}  -> "${section}" "${subsection}"
02391 //           ${Title}                       -> Nothing, variable expansion
02392 //
02393 //      3) __DBE_expression_list():
02394 //
02395 //           builds a vector of unbracketed whitespace separated strings, i.e.
02396 //
02397 //           "  ${Number}.a ${: Das Marmorbild} AB-${& Author= ${Eichendorf}-1870}"
02398 //
02399 //           is split into a vector
02400 //
02401 //              [0] ${Number}.a
02402 //              [1] ${: Das Marmorbild}
02403 //              [2] AB-${& Author= ${Eichendorf}}-1870
02404 //
02405 //           Each sub-expression is expanded using expand().
02406 //---------------------------------------------------------------------------
02407 inline std::string
02408 GetPot::__DBE_expand_string (const std::string str)
02409 {
02410     // Parses for closing operators '${ }' and expands them letting
02411     // white spaces and other letters as they are.
02412     std::string   new_string = "";
02413     unsigned open_brackets = 0;
02414     unsigned first = 0;
02415     unsigned i = 0;
02416     for (;  i < str.size(); ++i)
02417     {
02418         if ( i < str.size() - 2 && str.substr (i, 2) == "${" )
02419         {
02420             if ( open_brackets == 0 )
02421             {
02422                 first = i + 2;
02423             }
02424             open_brackets++;
02425         }
02426         else if ( str[i] == '}' && open_brackets > 0)
02427         {
02428             open_brackets -= 1;
02429             if ( open_brackets == 0 )
02430             {
02431                 const std::string Replacement = __DBE_expand (str.substr (first, i - first) );
02432                 new_string += Replacement;
02433             }
02434         }
02435         else if ( open_brackets == 0 )
02436         {
02437             new_string += str[i];
02438         }
02439     }
02440     return new_string;
02441 }

02443 inline STRING_VECTOR
02444 GetPot::__DBE_get_expr_list (const std::string str_, const unsigned ExpectedNumber)
02445 // ensures that the resulting vector has the expected number
02446 // of arguments, but they may contain an error message
02447 {
02448     std::string str = str_;
02449     // Separates expressions by non-bracketed whitespaces, expands them
02450     // and puts them into a list.
02451 
02452     unsigned i = 0;
02453     // (1) eat initial whitespaces
02454     for (; i < str.size(); ++i)
02455         if ( ! isspace (str[i]) )
02456         {
02457             break;
02458         }
02459 
02460     STRING_VECTOR   expr_list;
02461     unsigned         open_brackets = 0;
02462     std::vector<unsigned> start_idx;
02463     unsigned         start_new_string = i;
02464     unsigned         l = static_cast<unsigned int> (str.size() );
02465 
02466     // (2) search for ${ } expressions ...
02467     while ( i < l )
02468     {
02469         const char letter = str[i];
02470         // whitespace -> end of expression
02471         if ( isspace (letter) && open_brackets == 0)
02472         {
02473             expr_list.push_back (str.substr (start_new_string, i - start_new_string) );
02474             bool no_breakout_f = true;
02475             for (++i; i < l ; ++i)
02476             {
02477                 if ( ! isspace (str[i]) )
02478                 {
02479                     no_breakout_f = false;
02480                     start_new_string = i;
02481                     break;
02482                 }
02483             }
02484             if ( no_breakout_f )
02485             {
02486                 // end of expression list
02487                 if ( expr_list.size() < ExpectedNumber )
02488                 {
02489                     const std::string   pre_tmp ("<< ${ }: missing arguments>>");
02490                     STRING_VECTOR tmp (ExpectedNumber - expr_list.size(), pre_tmp);
02491                     expr_list.insert (expr_list.end(), tmp.begin(), tmp.end() );
02492                 }
02493                 return expr_list;
02494             }
02495         }
02496 
02497         // dollar-bracket expression
02498         if ( str.length() >= i + 2 && str.substr (i, 2) == "${" )
02499         {
02500             open_brackets++;
02501             start_idx.push_back (i + 2);
02502         }
02503         else if ( letter == '}' && open_brackets > 0)
02504         {
02505             int start = start_idx[start_idx.size() - 1];
02506             start_idx.pop_back();
02507             const std::string Replacement = __DBE_expand (str.substr (start, i - start) );
02508             if ( start - 3 < (int) 0)
02509             {
02510                 str = Replacement + str.substr (i + 1);
02511             }
02512             else
02513             {
02514                 str = str.substr (0, start - 2) + Replacement + str.substr (i + 1);
02515             }
02516             l = static_cast<unsigned int> (str.size() );
02517             i = start + static_cast<unsigned int> (Replacement.size() ) - 3;
02518             open_brackets--;
02519         }
02520         ++i;
02521     }
02522 
02523     // end of expression list
02524     expr_list.push_back (str.substr (start_new_string, i - start_new_string) );
02525 
02526     if ( expr_list.size() < ExpectedNumber )
02527     {
02528         const std::string   pre_tmp ("<< ${ }: missing arguments>>");
02529         STRING_VECTOR tmp (ExpectedNumber - expr_list.size(), pre_tmp);
02530         expr_list.insert (expr_list.end(), tmp.begin(), tmp.end() );
02531     }
02532 
02533     return expr_list;
02534 }

02586     else if ( expr[0] == '&' )

02593         {
02594             result += *it;
02595         }
02596

02601     else if ( expr.length() >= 3 && expr.substr (0, 3) == "<->" )
02602     {

02607         {

02640         STRING_VECTOR::const_iterator it = A.begin();
02641         double result = __convert_to_type (*it++, 0.0);

02651

02654         double result = __convert_to_type (*it++, 0.0);

02672     else if ( expr[0] == '^' )

02675         STRING_VECTOR::const_iterator it = A.begin();
02676         double result = __convert_to_type (*it++, 0.0);

02678         {
02679             result = std::pow (result, __convert_to_type (*it, 0.0) );
02680         }
02681         return __double2string (result);
02682     }
02683 
02684     // ${==  } ${<=  } ${>= } comparisons (return the number of the first 'match'
02685     else if ( expr.length() >= 2 &&
02686               ( expr.substr (0, 2) == "==" || expr.substr (0, 2) == ">=" ||
02687                 expr.substr (0, 2) == "<=" || expr[0] == '>'           || expr[0] == '<') )
02688     {
02689         // differentiate between two and one sign operators
02690         unsigned op = 0;
02691         enum { EQ, GEQ, LEQ, GT, LT };
02692         if      ( expr.substr (0, 2) == "==" )
02693         {
02694             op = EQ;
02695         }
02696         else if ( expr.substr (0, 2) == ">=" )
02697         {
02698             op = GEQ;
02699         }
02700         else if ( expr.substr (0, 2) == "<=" )
02701         {
02702             op = LEQ;
02703         }
02704         else if ( expr[0] == '>' )
02705         {
02706             op = GT;
02707         }
02708         else    /*                     "<" */
02709         {
02710             op = LT;
02711         }
02712 
02713         STRING_VECTOR a;
02714         if ( op == GT || op == LT )
02715         {
02716             a = __DBE_get_expr_list (expr.substr (1), 2);
02717         }
02718         else
02719         {
02720             a = __DBE_get_expr_list (expr.substr (2), 2);
02721         }
02722 
02723         std::string   x_orig = a[0];
02724         double   x = __convert_to_type (x_orig, 1e37);
02725         unsigned i = 1;
02726 
02727         STRING_VECTOR::const_iterator y_orig = a.begin();
02728         for (y_orig++; y_orig != a.end(); y_orig++)
02729         {
02730             double y = __convert_to_type (*y_orig, 1e37);
02731 
02732             // set the strings as reference if one wasn't a number
02733             if ( x == 1e37 || y == 1e37 )
02734             {
02735                 // it's a string comparison
02736                 if ( (op == EQ  && x_orig == *y_orig) || (op == GEQ && x_orig >= *y_orig) ||
02737                         (op == LEQ && x_orig <= *y_orig) || (op == GT  && x_orig >  *y_orig) ||
02738                         (op == LT  && x_orig <  *y_orig) )
02739                 {
02740                     return __int2string (i);
02741                 }
02742             }
02743             else
02744             {
02745                 // it's a number comparison
02746                 if ( (op == EQ  && x == y) || (op == GEQ && x >= y) ||
02747                         (op == LEQ && x <= y) || (op == GT  && x >  y) ||
02748                         (op == LT  && x <  y) )
02749                 {
02750                     return __int2string (i);
02751                 }
02752             }
02753             ++i;
02754         }
02755 
02756         // nothing fulfills the condition => return 0
02757         return "0";
02758     }
02759     // ${?? expr expr} select
02760     else if ( expr.length() >= 2 && expr.substr (0, 2) == "??" )
02761     {
02762         STRING_VECTOR a = __DBE_get_expr_list (expr.substr (2), 2);
